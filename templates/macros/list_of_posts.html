{{- article_cards(articles=ns.featured_posts) -}}
{% set ns.featured_posts = [] %} {% if ns.remaining_posts|count >= 3 %}
{{- post_links(posts=ns.remaining_posts) -}}
{% set ns.remaining_posts = [] %} {% endif %} {% endfor %}
